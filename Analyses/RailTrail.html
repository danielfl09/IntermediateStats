# Step 3
get_upper_tri <- function(data_set){
  data_set[lower.tri(data_set)] <- NA
  return(data_set)
}

upper_tri_trai <- get_upper_tri(cormat_trail)
melted_upper_tri <- melt(upper_tri_trai, na.rm = TRUE)
